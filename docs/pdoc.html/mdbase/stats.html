Statistical calculations and plots in MDBASE.

  • In MDBASE, statistical calculations are closely connected with plotting.
  • MDBASE calculates 4 different plot types + corresponding statistics:
    • Correlation plot + statistics: Pearson's r, p-values, R2 coefficients
    • Scatterplot matrix graph + statistics: Pearson's r + p-values
    • Scatterplot matrix table in the form of heatmap showing Pearson's r
    • Boxplots + statistics: p-values quantifying differences among groups

101    def regression(self, P1, P2, category=None, 
102            rtype='linear', marker='k-', label=None,
103            r_to_plot=None, p_to_plot=None, R2_to_plot=None):
104        '''
105        Calculate regression and add regression curve to graph/plot.
106        
107        * Regression = fitting of pre-defined functions to XY-data.
108        * Pre-defined function is selected according to rtype argument.
109        * XY-data = P1,P2 = X-data, Y-data.
110        * Statistical coefficients can be added to the graph as well.
111
112        Parameters
113        ----------
114        P1 : list or array
115            X-values for regression.
116        P2 : list or array
117            Y-values for regression.
118        category : bool, optional, the default is None
119            Advanced option.
120            The correlationPlot object (self)
121            contains this function (self.regression)
122            and also reference to data (self.df) = ref to pandas DataFrame.
123            If category is given,
124            only self.df[category] are employed in regression.
125            Example: To plot all data, but fit only TKR data,
126            the regression function can be called as
127            `CorrelationPlot.regression(P1,P2,category=(df.Itype==THR)`.
128        rtype : str, optional, default is 'linear'
129            Regression type.
130            Pre-defined function for regression.
131            It can be one of: 'linear', 'linear_kx', 'quadratic', 'power'.
132        marker : matplotlib format string, optional, the default is 'k-'
133            Matplotlib format string defines the color and type of line/points.
134            More details: GoogleSearch - matplotlib.pyplot.plot.
135        label : str, optional, default is None
136            Label that can be assigned to regression curve.
137            If defined, this label will appear in the legend of the plot.
138        r_to_plot : (float,float,int), optional, the default is None
139            If r_to_plot contains list with three numbers (x,y,n),
140            value of Pearson's r is added to graph/plot
141            at position (x,y), rounded to (n) decimals.
142        p_to_plot : (float,float,int), optional, the default is None
143            If p_to_plot contains list with three numbers (x,y,n),
144            p-value is added to graph/plot
145            at position (x,y), rounded to (n) decimals.
146        R2_to_plot : (float,float,int), optional, the default is None
147            If R2_to_plot contains list with three numbers (x,y,n),
148            R2 coefficient is added to graph/plot
149            at position (x,y), rounded to (n) decimals.
150            
151        Returns
152        -------
153        None
154            The output is the regression/fitting curve
155            added to the self.ax object.
156            Additional output are statistical coefficients
157            printed to stdout.
158            Yet additional (optional) output are statistical
159            coefficients added to self.ax object.
160        '''
161        # Test, if category was given
162        # (trick: pandas.Series cannot be compared with scalar => np.any
163        if np.any(category) == None:
164            ds = self.df
165        else:
166            ds = self.df[category]
167        # Remove NaN values before regression
168        # (scipy.optimize.curve_fit does not work with NaN's
169        ds = ds[[P1,P2]]
170        ds = ds.dropna()
171        # Sort values
172        ds = ds.sort_values(by=[P1,P2])
173        # Get regression type & define regression function
174        if rtype == 'linear':
175            def linear(X,a,b): return(a*X + b)
176            self.regression_func = linear
177            self.regression_func_eq = 'y = a*x + b'
178        elif rtype == 'linear_kx':
179            def linear_kx(X,k): return(k*X)
180            self.regression_func = linear_kx
181            self.regression_func_eq = 'y = k*x'
182        elif rtype == 'quadratic':
183            def quadratic(X,a,b,c): return(a*X**2 + b*X + c)
184            self.regression_func = quadratic
185            self.regression_func_eq = 'y = ax**2 + b*x + c'
186        elif rtype == 'power':
187            def power(X,n,c): return(c*X**n)
188            self.regression_func = power
189            self.regression_func_eq = 'y = c * x**n'
190        else:
191            print('Unknown regression type (rtype) - no action.')    
192        # Calculate regression
193        X,Y = (ds[P1],ds[P2])
194        par,cov = scipy.optimize.curve_fit(self.regression_func,X,Y)
195        # Calculate Pearsons's coefficient r and p-value
196        r_coeff,p_value = scipy.stats.pearsonr(X,Y)
197        # Calculate coefficient of determination
198        # R2 values: 1 = perfect, 0 = estimate~average(Y), negative = very bad 
199        # https://en.wikipedia.org/wiki/Coefficient_of_determination
200        Yave = np.average(Y)
201        Yfit = self.regression_func(X,*par)
202        SSres = np.sum((Y-Yfit)**2)
203        SStot = np.sum((Y-Yave)**2)
204        R2 = 1 - SSres/SStot
205        # Show regression in graph
206        self.ax.plot(X,self.regression_func(X,*par), marker, label=label)
207        # Add statistical parameters to graph if required
208        if r_to_plot != None: self.add_coeff_to_plot(r_coeff,r'$r$',*r_to_plot)
209        if p_to_plot != None: self.add_coeff_to_plot(p_value,r'$p$',*p_to_plot)
210        if R2_to_plot != None: self.add_coeff_to_plot(R2,r'$R^2$',*r_to_plot)
211        # Print statistics to stdout
212        self.print_statistics_to_stdout(par, r_coeff, p_value, R2)

301    def _pairplot_statistics(self, r_coeffs, p_values,
302                             r_pos, p_pos, r_decimals, p_decimals):
303        '''
304        Draw values of r-coefficients and p-values to the pairplot.
305
306        Parameters
307        ----------
308        r_coeffs : bool
309            If True, Pearson correlation coefficients r are calculated.
310            The Pearson r range from +1 (perfect positive correlation)
311            through 0 (no correlation) to -1 (perfect negative correlation). 
312        p_values : bool
313            If True, p-values are calculated.
314            In this case, the p-values represent the probability
315            that we would get such a strong (or stronger) correlation
316            just by coincidence.
317        r_pos : (float,float)
318            Coordinates for drawing the r coefficient in the graph.
319        p_pos : (float,float)
320            Coordinates for drawing the p-values in the graph.
321        r_decimals : int
322            The number of decimals for r coefficients.
323        p_decimals : int
324            The number of decimals for p-values.
325
326        Returns
327        -------
328        * None; the output are just the p-coeff and r-values in the pairplot.
329        '''
330        # Scipy + matplotlib: add Pearson's r to upper-left corners
331        # Trick #0: for k in dict: for-cycle with dictionary returns keys
332        # Trick #1: for i,k in enumerate(dict): trick0+enumerate => index+key
333        # Trick #2: ax.text(x,y,s,transform=ax.transAxes) rel.coords ~ (0;1)
334        # Trick #3: ax.text(*pos,...) => x,y coordinates in pos are unpacked
335        # (*list and **dict = Python unpacking operators, for function calls
336        for index1,column1 in enumerate(self.properties.keys()):
337            for index2,column2 in enumerate(self.properties.keys()):
338                (corr,pval) = scipy.stats.pearsonr(
339                    self.df[column1],self.df[column2])
340                if column1 != column2:
341                    if r_coeffs == True:
342                        self.grid.axes[index1,index2].text(
343                            *r_pos, f'$r$ = {corr:.{r_decimals}f}',
344                            transform=self.grid.axes[index1,index2].transAxes)
345                    if p_values == True:
346                        self.grid.axes[index1,index2].text(
347                            *p_pos, f'$p$ = {pval:.{p_decimals}f}',
348                            transform=self.grid.axes[index1,index2].transAxes)

382        fig1,ax1 = plt.subplots()

389    def draw(self, 
390             cmap_r='Reds', cmap_p='Blues_r', 
391             decimals_r=2, decimals_p=2, cbars=True):
392        # (1) Prepare data for calculations
393        ds = self.df[list(self.properties.keys())]
394        ds = ds.dropna()
395        # (2) Prepare empty tables to save results
396        n = len(self.properties)
397        r_values = np.zeros((n,n))
398        p_values = np.zeros((n,n))
399        # (3) Calculate correlations
400        for (i,column1) in enumerate(self.properties.keys()):
401            for (j,column2) in enumerate(self.properties.keys()):
402                (corr,pval) = scipy.stats.pearsonr(ds[column1],ds[column2])
403                r_values[i,j] = round(corr,5)
404                p_values[i,j] = round(pval,5)
405        # (4) Prepare for plotting... 
406        # (Flip rows so that the main diagonal started in upper left corner
407        # (default: [0,0] = start of the main diagonal = lower left corner
408        r_values = np.flipud(r_values)
409        p_values = np.flipud(p_values)
410        # (5a) Draw cmatrix for r-values = sns.heatmap
411        # ...draw cmatrix = heatmap
412        my_format = "."+str(decimals_r)+"f"
413        sns.heatmap(data=r_values, ax=self.ax1,
414            annot=True, fmt=my_format, cmap=cmap_r, cbar=cbars,
415            linecolor='white', linewidth=2)
416        # (5b) Draw cmatrix for p-values = sns.heatmap with custom colormap
417        # ...draw cmatrix = heatmap
418        my_format = "."+str(decimals_p)+"f"
419        sns.heatmap(data=p_values, ax=self.ax2,
420            annot=True, fmt=my_format, cmap=cmap_p, cbar=cbars,
421            linecolor='white', linewidth=2)

427        my_yticklabels = list(reversed(list(self.properties.values())))

434        # Final adjustments
435        for fig in (self.fig1,self.fig2):
